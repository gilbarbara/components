import { Meta } from '@storybook/addon-docs';
import { capitalize } from '@gilbarbara/helpers';
import { hex2hsl, textColor } from 'colorizr';

import { Box, FlexCenter, Grid, H2, H3, Jumbo, Paragraph, Text } from 'src';
import { colors, grayScale, variants as themeVariants } from 'src/modules/theme';
import { shades, variants } from '../__helpers__';

<Meta title="Overview/Colors" />

<Jumbo>Colors and shades</Jumbo>

<H2>Base colors</H2>

<Grid gap={32} mb="xxl" justifyContent="start" templateColumns="repeat(5, auto)">
  {Object.entries(colors)
    .map(([key, value]) => {
      const { h, s, l } = hex2hsl(value);
      return (
        <Box key={key}>
          <FlexCenter
            height={128}
            radius="round"
            shade="mid"
            variant={key}
            width={128}
          >
            {key}
          </FlexCenter>
          <Paragraph align="center" size="small" mt="xs">HEX: {value}</Paragraph>
          <Paragraph align="center" size="small">
            H: {Math.round(h)} / S: {Math.round(s)} / L: {Math.round(l)}
          </Paragraph>
        </Box>
      );
    })}
</Grid>

<H2>Variants</H2>

<Box mb="xxl">
  {Object.entries(themeVariants)
    .map(([variant, color]) => (
      <Box key={variant} mb="lg">
        <H3 light>{capitalize(variant)}</H3>
        <Grid gap={32} justifyContent="start" templateColumns="repeat(7, auto)">
          {Object.entries(color)
            .map(([shade, value]) => {
              const { h, s, l } = hex2hsl(value.bg);
              return (
                <Box key={shade}>
                  <FlexCenter
                    height={128}
                    radius="round"
                    shade={shade}
                    variant={variant}
                    width={128}
                  >
                    {shade}
                  </FlexCenter>
                  <Paragraph align="center" size="small" mt="xs">HEX: {value.bg}</Paragraph>
                  <Paragraph align="center" size="small">
                    H: {Math.round(h)} / S: {Math.round(s)} / L: {Math.round(l)}
                  </Paragraph>
                </Box>
              );
            })}
        </Grid>
      </Box>
    ))}
</Box>


<H2>Shades</H2>

<Box mb="xxl">
  {shades
    .map((shade) => (
      <Box key={shade} mb="lg">
        <H3 light>{capitalize(shade)}</H3>
        <Grid gap={32} justifyContent="start" templateColumns="repeat(5, auto)">
          {variants
            .map((variant) => {
              const { bg } = themeVariants[variant][shade];
              const { h, s, l } = hex2hsl(bg);
              return (
                <Box key={variant}>
                  <FlexCenter
                    height={128}
                    radius="round"
                    shade={shade}
                    variant={variant}
                    width={128}
                  >
                    {variant}
                  </FlexCenter>
                  <Paragraph align="center" size="small" mt="xs">HEX: {bg}</Paragraph>
                  <Paragraph align="center" size="small">
                    H: {Math.round(h)} / S: {Math.round(s)} / L: {Math.round(l)}
                  </Paragraph>
                </Box>
              );
            })}
        </Grid>
      </Box>
    ))}
</Box>

<H2>Grayscale</H2>

<Grid gap={32} justifyContent="start" templateColumns="repeat(5, auto)">
  {Object.entries(grayScale)
    .map(([key, color]) => (
      <Box key={color}>
        <FlexCenter
          height={128}
          radius="round"
          width={128}
          style={{ backgroundColor: color }}
        >
          <Text style={{ color: textColor(color) }}>{color}</Text>
        </FlexCenter>
        <Paragraph align="center" size="small" mt="xs">{key}</Paragraph>
      </Box>
    ))}
</Grid>
